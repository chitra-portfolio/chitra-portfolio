        function filterProjects(category) {

            const cards = document.querySelectorAll('.flip-card');

            const buttons = document.querySelectorAll('.filter-btn');

            

            buttons.forEach(btn => btn.classList.remove('active'));

            event.target.classList.add('active');

            

            cards.forEach(card => {

                card.classList.remove('flipped');

                card.classList.remove('enlarged');

                if (category === 'all') {

                    card.style.display = 'block';

                } else {

                    const categories = card.getAttribute('data-category').split(' ');

                    if (categories.includes(category)) {

                        card.style.display = 'block';

                    } else {

                        card.style.display = 'none';

                    }

                }

            });

            document.querySelector('.flip-overlay').classList.remove('active');

        }


        function enlargeCard(card) {

            const overlay = document.querySelector('.flip-overlay');

            const allCards = document.querySelectorAll('.flip-card');

            

            allCards.forEach(c => {

                if (c !== card) {

                    c.classList.remove('enlarged');

                    c.classList.remove('flipped');

                }

            });

            

            card.classList.toggle('enlarged');

            card.classList.toggle('flipped');

            overlay.classList.toggle('active');

        }


        function closeEnlarged() {

            document.querySelectorAll('.flip-card').forEach(card => {

                card.classList.remove('enlarged');

                card.classList.remove('flipped');

            });

            document.querySelector('.flip-overlay').classList.remove('active');

        }


        function toggleAboutSection(section) {

            const details = document.getElementById(section);

            const icon = document.getElementById(section + '-icon');

            details.classList.toggle('open');

            icon.classList.toggle('open');

        }


        let currentTestimonialSet = 0;

        const totalTestimonialSets = 4;


        function moveCarousel(direction) {

            const sets = document.querySelectorAll('.testimonial-set');

            sets[currentTestimonialSet].classList.remove('active');

            

            currentTestimonialSet += direction;

            if (currentTestimonialSet < 0) currentTestimonialSet = totalTestimonialSets - 1;

            if (currentTestimonialSet >= totalTestimonialSets) currentTestimonialSet = 0;

            

            sets[currentTestimonialSet].classList.add('active');

        }
